    <form id="calculatorForm">

        <!-- Labor Calculator -->

        <label for="ideal_labor">Enter your ideal labor hours:</label>

        <input type="number" id="ideal_labor" required><br>

        

        <label for="actual_labor">Enter your actual labor hours:</label>

        <input type="number" id="actual_labor" required><br>


        <!-- Beef Efficiency Calculator -->

        <label for="ideal_beef">Enter your ideal beef number:</label>

        <input type="number" id="ideal_beef" required><br>

        

        <label for="actual_beef">Enter your actual number of beef:</label>

        <input type="number" id="actual_beef" required><br>


        <button type="button" onclick="calculateLaborAndBeef()">Calculate</button>

    </form>

    

    <p id="laborResult"></p>

    <p id="beefResult"></p>


    <script>

        function calculateLaborAndBeef() {

            // Labor Calculator

            var ideal_labor = parseFloat(document.getElementById("ideal_labor").value);

            var actual_labor = parseFloat(document.getElementById("actual_labor").value);

            var labor_difference = ideal_labor - actual_labor;

            var laborResultElement = document.getElementById("laborResult");


            if (labor_difference < 0) {

                laborResultElement.textContent = "Watch your labor. You exceeded the ideal by " + (-labor_difference) + " hours.";

            } else if (labor_difference == 0) {

                laborResultElement.textContent = "You met the ideal labor hours.";

            } else {

                laborResultElement.textContent = "Saving labor. You saved " + labor_difference + " hours.";

            }


            // Beef Efficiency Calculator

            var ideal_beef = parseFloat(document.getElementById("ideal_beef").value);

            var actual_beef = parseFloat(document.getElementById("actual_beef").value);

            var beef_efficiency = (actual_beef / ideal_beef) * 100;

            var beefResultElement = document.getElementById("beefResult");


            beefResultElement.textContent = "Your beef efficiency is: " + beef_efficiency.toFixed(2) + "%";


            if (beef_efficiency >= 98) {

                beefResultElement.textContent += " Beef efficiency is good!";

            } else {

                beefResultElement.textContent += " Beef efficiency needs improvement.";

            }

        }

    </script>
